DROP TABLE IF EXISTS wp_tsa_skater_realtime;

CREATE TABLE wp_tsa_skater_realtime (
    gameDate DATE,
    gameId BIGINT,
    skaterFullName VARCHAR(150),
    playerId BIGINT,
    homeRoad VARCHAR(10),
    teamAbbrev VARCHAR(10),
    opponentTeamAbbrev VARCHAR(10),
    shootsCatches VARCHAR(10),
    positionCode VARCHAR(10),
    gamesPlayed INT,

    timeOnIcePerGame DECIMAL(10,4),
    hits INT,
    hitsPer60 DECIMAL(10,4),
    blockedShots INT,
    blockedShotsPer60 DECIMAL(10,4),
    giveaways INT,
    giveawaysPer60 DECIMAL(10,4),
    takeaways INT,
    takeawaysPer60 DECIMAL(10,4),
    firstGoals INT,
    otGoals INT,
    emptyNetGoals INT,
    emptyNetAssists INT,
    emptyNetPoints INT,
    totalShotAttempts INT,
    shotAttemptsBlocked INT,
    missedShots INT,
    missedShotWideOfNet INT,
    missedShotOverNet INT,
    missedShotGoalpost INT,
    missedShotCrossbar INT,
    missedShotShort INT,
    missedShotFailedBankAttempt INT,

    lastName VARCHAR(100),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_playerId (playerId),
    INDEX idx_teamAbbrev (teamAbbrev),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_positionCode (positionCode),
    INDEX idx_homeRoad (homeRoad),
    INDEX idx_skaterFullName (skaterFullName)
);

DELETE FROM wp_tsa_skater_realtime
WHERE skaterFullName = 'skaterFullName';
