DROP TABLE IF EXISTS wp_tsa_skater_summaryshooting;

CREATE TABLE wp_tsa_skater_summaryshooting (
    gameDate DATE,
    gameId BIGINT,
    skaterFullName VARCHAR(150),
    playerId BIGINT,
    homeRoad VARCHAR(10),
    opponentTeamAbbrev VARCHAR(10),
    shootsCatches VARCHAR(10),
    positionCode VARCHAR(10),
    gamesPlayed INT,

    timeOnIcePerGame5v5 DECIMAL(10,4),

    satFor INT,
    satAgainst INT,
    satTotal INT,
    satAhead INT,
    satTied INT,
    satBehind INT,
    satClose INT,
    satRelative DECIMAL(10,4),

    usatFor INT,
    usatAgainst INT,
    usatTotal INT,
    usatAhead INT,
    usatTied INT,
    usatBehind INT,
    usatClose INT,
    usatRelative DECIMAL(10,4),

    lastName VARCHAR(100),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_playerId (playerId),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_positionCode (positionCode),
    INDEX idx_homeRoad (homeRoad),
    INDEX idx_skaterFullName (skaterFullName)
);

DELETE FROM wp_tsa_skater_summaryshooting
WHERE skaterFullName = 'skaterFullName';
